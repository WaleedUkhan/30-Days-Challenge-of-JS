Session Storage Lecture

// Function to save a string value to sessionStorage function saveToSessionStorage(key, value) { sessionStorage.setItem(key, value); console.log(`Saved: ${key} = ${value}`); } // Function to retrieve a value from sessionStorage and log it function retrieveFromSessionStorage(key) { const value = sessionStorage.getItem(key); console.log(`Retrieved: ${key} = ${value}`); return value; } // Example usage const key = "exampleKey"; const value = "Hello, sessionStorage!"; // Save to sessionStorage saveToSessionStorage(key, value); // Retrieve from sessionStorage and log the value retrieveFromSessionStorage(key);
